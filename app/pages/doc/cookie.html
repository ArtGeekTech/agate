cookie

koa已经整合了cookie功能,是基于https://github.com/pillarjs/cookies这个模块实现的

我们可以在控制器上这样访问的所有cookies

//pages/doc/controller.js
exports.cookie = function *(next) {
   var cookie = this.request.header.cookie
   yield this.render("doc/index", {
       cookie: cookie
   })
}

设置cookie

exports.cookie = function *(next) {
    this.cookies.set('aaa', 'bbb' ); 
    this.cookies.set('xxx','yyy' );
   yield this.render("doc/index")
}

获取cookie

exports.cookie = function *(next) {
   var a = this.cookies.get('aaa')
   yield this.render("doc/index",{
      aaa: a
   })
}
